+ [7453번 합이 0인 네 정수](https://www.acmicpc.net/problem/7453)
    <pre>
    <code>
    import sys
    read = sys.stdin.readline

    n = int(read())
    A,B,C,D=[],[],[],[]
    for _ in range(n):
        a,b,c,d = map(int,read().split())
        A.append(a);B.append(b);C.append(c);D.append(d)

    ab={}

    for a in A:
        for b in B:
            ab[a+b] = ab.get(a+b,0)+1

    ans = 0

    for c in C:
        for d in D:
            ans+=ab.get(-(c+d),0)

    print(ans)
    </code>
    </pre>
