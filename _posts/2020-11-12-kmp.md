# KMP 알고리즘

## O(N)

    import sys

    read= sys.stdin.readline

    T= read().rstrip()
    P= read().rstrip()

    def fail(P):  # 실패함수. 접미사. 접두사 일치하는 길이 배열에 저장후 리턴.
        len_s = len(P)
        f = [0]*len_s
        begin =1
        m=0
        
        while begin + m < len_s : 
            if P[begin+m] == P[m] : # 
                m+=1
                f[begin+m-1]=m;
            else:
                if m==0:
                    begin+=1
                else:
                    begin+=(m-f[m-1])
                    m=f[m-1]
        return f


    def kmp():
        len_h = len(T)
        len_s = len(P)

        f = fail(P)
        r = []

        begin =0
        m=0 # 일치 개수.
        while begin<= len_h - len_s : 
            if m<len_s and T[begin+m]==P[m]: # m 이 P의 길이보다 작고 지금까지는 일치함.
                m+=1
                if(m==len(P)):
                    r.append(begin+1)
            else: #불일치하거나 p를 찾은 경우 
                if m ==0: # 일치한적 없음. 
                    begin+=1
                else: # 다음 위치를 옮김.
                    begin+= (m-f[m-1])
                    m=f[m-1]
        return r

    answer = kmp()
    print(len(answer))
    print(*answer)
